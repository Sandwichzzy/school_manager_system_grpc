    --------------- BUILDING API BINARY
    #参数	作用
    #GOOS=darwin	目标操作系统是 macOS
    #GOARCH=amd64	目标架构是 x86_64（Intel Mac）
    #-o macOS_api_x86-64.exe	输出文件名
    #server.go	编译入口文件

    #跨平台编译（GOOS / GOARCH）
    GOOS=darwin GOARCH=amd64 go build -o macOS_api_x86-64 server.go (Not if you are working on macOS amd64 system)
    GOOS=windows GOARCH=amd64 go build -o windows_binary/binary-windows_x86-64 server.go (Not if you are working on macOS amd64 system)
